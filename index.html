Author

Владимир Кретов

Работа посвящена анализу стилистических особенностей прозы ряда авторов XVIII–XIX веков: Джейн Остин (JA), Энн Бронте (AB), Шарлотта Бронте (CB), Эмили Бронте (EB), Чарльз Диккенс (CD), Джордж Элиот (GE), Элизабет Гаскелл (GE), Генри Филдинг (HF), Сэмюэл Ричардсон (SR), Лоренс Стерн (LS), Уильям Теккерей (WT), Энтони Троллоп (AT).

Основная задача исследования — построить и обучить предиктивную модель машинного обучения, которая способна автоматически определять авторство текстового фрагмента на основе его количественных языковых характеристик.

metadata_fixed <- metadata |> 
  mutate(
    textID = as.numeric(str_extract(`textID author`, "^\\d+")),
    author = authorID,
    authorID = title,
    title = `1stPubl`,
    `1stPubl` = as.numeric(author_gender),
    
    # Заполняем author_gender с нуля на основе имени автора
    author_gender = case_when(
      author %in% c("Austen, Jane", "Eliot, George", "Gaskell, Elizabeth", 
                    "Bronte, Charlotte", "Bronte, Emily", "Bronte, Anne", 
                    "Woolf, Virginia", "Shelley, Mary") ~ 1, # Женщины

Большинство ключевых авторов (Antony Trollope, William Makepeace Thackeray, George Eliot, Charles Dickens, Charlotte Brontë, Jane Austen) представлены равным объемом — по три романа на каждого. Дисбаланс классов подтверждает необходимость нарезки текстов на чанки: этот шаг позволит искусственно размножить выборку и нивелировать дисбаланс при обучении моделей.

Показать код 
# Создаем финальную таблицу слов для графиков и анализа
eda_words <- dataset |> 
  # Очистка структуры: удаляем слово chapter и любой идущий за ним маркер главы
  mutate(
    text = str_remove_all(text, "(?i)\\bchapter\\b\\s*([a-z0-9]+|the\\s+[a-z]+)?")
  ) |> 
  # Токенизация
  unnest_tokens(word, text) |> 
  # Очистка от цифр
  filter(!str_detect(word, "^\\d+$")) |> 
  # Дополнительная фильтрация шума
  filter(str_length(word) > 1 | word %in% c("i", "a")) |> 
  # Производим лемматизацию
  mutate(lemma = lemmatize_words(word))

# Считаем TF-IDF (без удаления стоп-слов)
word_counts_tfidf <- eda_words |>
  count(author, word, sort = TRUE) |>
  bind_tf_idf(word, author, n)

# Строим график
top_words_tfidf <- word_counts_tfidf |> 
  group_by(author) |> 
  slice_max(tf_idf, n = 10, with_ties = FALSE) |> 
  ungroup() |> 
  ggplot(aes(
    x = tf_idf, 
    y = reorder_within(word, tf_idf, author), 
    fill = author
  )) +
  geom_col(show.legend = FALSE) +
  facet_wrap(~ author, scales = "free_y", ncol = 3) +
  scale_y_reordered() +
  theme_minimal(base_size = 11) +
  theme(
    strip.text = element_text(face = "bold", size = 10),
    panel.spacing = unit(1.5, "lines"),
    plot.title = element_text(face = "bold", size = 14, hjust = 0.5)
  ) +
  labs(
    title = "Топ-10 слов по метрике TF-IDF (со стоп-словами)",
    x = "Значение TF-IDF",
    y = NULL
  )

print(top_words_tfidf)

График выше демонстрирует фундаментальное свойство меры TF-IDF в текстовом анализе. Хотя на этапе предобработки не удалялись стоп-слова, алгоритм успешно проигнорировал их: они распределены по всему корпусу равномерно и имеют минимальный информационный вес. В результате в топы уникальных маркеров для каждого автора вышли исключительно омонимы — имена ключевых персонажей романов, такие как Элинор и Эмма у Джейн Остин, Хитклифф у Эмили Бронте или Ловелас у Сэмюэла Ричардсона, а также характерные топонимы вроде Торнфилда у Шарлотты Бронте.

Этот график отлично отражает совокупность уникальных именованных сущностей каждого писателя, однако для построения обобщающей модели классификации такой результат не подходит, поскольку алгоритм будет банально запоминать сюжеты и имена героев конкретных книг вместо выявления скрытых грамматических и стилистических паттернов самого автора.

Метод чистого подсчета частоты (TF) выводит на первый план лексическое ядро, общее для большинства произведений викторианской эпохи. Вместо уникальных имен собственных — маркеры быта, времени и социального этикета того периода: time, day, hand, house, miss, lady, sir.

Тем не менее специфика индивидуальных сюжетов все еще пробивается у авторов с небольшим количеством текстов в корпусе. Ярким примером служит Эмили Бронте, в топе у которой из-за единственного романа «Грозовой перевал» удержались знаковые имена heathcliff и linton. Аналогично у Лоренса Стерна выделяются слова uncle и toby, привязанные к его главному произведению.

Простое удаление стоп-слов оставляет модель уязвимой к тематическому шуму конкретных книг. Это подтверждает необходимость дальнейшего перехода к фиксированному пространству из 500 самых частотных слов корпуса.

Показать код 
# Задаем размер чанка
chunk_size <- 5000

dataset_chunks <- dataset |> 
  unnest_tokens(word, text) |> 
  group_by(textID) |> 
  mutate(chunk_id = (row_number() - 1) %/% chunk_size + 1) |> 
  group_by(textID, authorID, author, author_gender, title, chunk_id) |> 
  summarise(text = paste(word, collapse = " "), .groups = "drop") |> 
  mutate(authorID = as.factor(authorID))

# Посмотрим, сколько чанков у нас получилось суммарно

Разобьем выборку на обучающую и тестовую части.

Показать код 
set.seed(42)
# Делаем разбиение (80% на обучение, 20% на тест)
chunk_split <- initial_split(dataset_chunks, prop = 0.80, strata = authorID)
train_data <- training(chunk_split)
test_data  <- testing(chunk_split)

В четвертом блоке кода создаются две модели для сравнения. Сначала оформляется рецепт обработки текста, который берет чанки и превращает их в числа. Он разбивает текст на отдельные слова, отбирает 500 самых частотных и считает их относительную частоту в каждом чанке. В этот топ явно попадут служебные части речи, так что модель будет анализировать стиль авторов, а не сюжеты книг.

Затем создаются спецификации двух алгоритмов классификации. Первый — логистическая регрессия, которая ищет линейные зависимости между частотой слов. Второй — случайный лес из 500 деревьев решений, способный находить сложные нелинейные сочетания признаков.

Показать код 
style_recipe <- recipe(authorID ~ text, data = train_data) |> 
  step_tokenize(text) |> 
  step_tokenfilter(text, max_tokens = 500) |> 
  step_tf(text, weight_scheme = "term frequency")

# Логистическая регрессия
logreg_spec <- parsnip::multinom_reg(penalty = 0.1, mixture = 1) |> 
  parsnip::set_engine("glmnet") |> 
  parsnip::set_mode("classification")

logreg_workflow <- workflows::workflow() |> 
  workflows::add_recipe(style_recipe) |> 
  workflows::add_model(logreg_spec)

# Случайный лес
rf_spec <- parsnip::rand_forest(trees = 500) |> 
  parsnip::set_engine("ranger", importance = "impurity") |> 
  parsnip::set_mode("classification")

rf_workflow <- workflows::workflow() |> 
  workflows::add_recipe(style_recipe) |> 
  workflows::add_model(rf_spec)

В абсолютном большинстве случаев модель демонстрирует великолепную чувствительность, присваивая чанкам истинных авторов экстремально высокие вероятности.
